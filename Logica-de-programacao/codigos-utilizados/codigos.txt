// Laços de repetição

//do while - faça enquanto - primeiro faz depois verifica
let i = 3

do {
    console.log("ola")
    i++
} while(i < 3)

//resultado vai ser 1 vez "ola" no console

//while - enquanto uma condição for verdadeira, ele executa o bloco de código

    console.log("ola")

}
